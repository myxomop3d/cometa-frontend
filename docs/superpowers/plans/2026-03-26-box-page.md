
# Box Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a `/box` page with composable filter components (text, number range, date range, checkbox, select, relation dropdown/modal) and inline editing, refactoring SimpleTable to use a `filterSlot` prop.

**Architecture:** Composable filter components that each manage their own UI and report values via `onChange`. SimpleTable receives a `filterSlot` React node instead of building filters internally. The Box page composes all filters into a `FilterBar` grid.

**Tech Stack:** React, TypeScript, TanStack Router/Query/Table, shadcn/ui (dialog, popover, badge, calendar), Zod, MSW, Tailwind CSS v4

```ts
/** Resolves a potentially nested path like "item/id" on an object */
function resolveField(rec: Record<string, unknown>, field: string): unknown {
  const parts = field.split("/");
  let current: unknown = rec;
  for (const part of parts) {
    if (current == null || typeof current !== "object") return undefined;
    current = (current as Record<string, unknown>)[part];
  }
  return current;
}
```

Replace the entire `matchCondition` function with:

```ts
function matchCondition<T extends object>(
  item: T,
  condition: string,
): boolean {
  const rec = item as Record<string, unknown>;

  // any(alias: innerConditions) — collection lambda
  // e.g. things/any(t: t/id eq 1 or t/id eq 3)
  // e.g. tags/any(t: contains_ignoring_case(t,'v'))
  const anyMatch = condition.match(/^([\w]+)\/any\((\w+):\s*(.+)\)$/i);
  if (anyMatch) {
    const [, collection, alias, innerExpr] = anyMatch;
    const arr = rec[collection];
    if (!Array.isArray(arr)) return false;
    // Split inner expression by " or "
    const orConditions = innerExpr.split(/\s+or\s+/i);
    return arr.some((element) => {
      // For primitive arrays (like string[]), the alias IS the element
      // For object arrays, the alias is a placeholder prefix (e.g. t/id -> element.id)
      return orConditions.some((orCond) => {
        // Replace alias references with actual traversal
        const resolved = orCond.trim();
        // Check if this is a function call with the alias as argument
        // e.g. contains_ignoring_case(t,'v') where t is a primitive
        const funcWithAlias = resolved.match(
          /^(contains(?:_ignoring_case)?)\((\w+),\s*'([^']*)'\)$/i
        );
        if (funcWithAlias && funcWithAlias[2] === alias) {
          const val = String(element ?? "").toLowerCase();
          return val.includes(funcWithAlias[3].toLowerCase());
        }
        // Otherwise, handle alias/field patterns like "t/id eq 5"
        const eqParts = resolved.match(
          /^(\w+)\/(\w+)\s+(eq|ge|le|gt|lt)\s+(.+)$/i
        );
        if (eqParts && eqParts[1] === alias) {
          const field = eqParts[2];
          const op = eqParts[3].toLowerCase();
          const rawVal = eqParts[4];
          const fieldVal =
            typeof element === "object" && element != null
              ? (element as Record<string, unknown>)[field]
              : undefined;
          const compareVal = parseODataValue(rawVal);
          return compareValues(fieldVal, op, compareVal);
        }
        return true;
      });
    });
  }

  // contains(field,'value') or contains_ignoring_case(field,'value')
  // field can be a nested path like item/name
  const containsMatch = condition.match(
    /^contains(?:_ignoring_case)?\(([\w/]+),\s*'([^']*)'\)$/i
  );
  if (containsMatch) {
    const [, field, value] = containsMatch;
    const fieldVal = String(resolveField(rec, field) ?? "").toLowerCase();
    return fieldVal.includes(value.toLowerCase());
  }

  // startswith(field,'value')
  const startsMatch = condition.match(
    /^startswith\(([\w/]+),\s*'([^']*)'\)$/i
  );
  if (startsMatch) {
    const [, field, value] = startsMatch;
    const fieldVal = String(resolveField(rec, field) ?? "").toLowerCase();
    return fieldVal.startsWith(value.toLowerCase());
  }

  // field op value — handles eq, ge, le, gt, lt with quoted strings, numbers, booleans
  const opMatch = condition.match(
    /^([\w/]+)\s+(eq|ge|le|gt|lt)\s+(.+)$/i
  );
  if (opMatch) {
    const [, field, op, rawVal] = opMatch;
    const fieldVal = resolveField(rec, field);
    const compareVal = parseODataValue(rawVal);
    return compareValues(fieldVal, op.toLowerCase(), compareVal);
  }

  return true;
}
```

- [ ] **Step 3: Add `parseODataValue` and `compareValues` helpers**

Add after the `resolveField` function:

```ts
/** Parses an OData literal: 'string', number, true/false */
function parseODataValue(raw: string): string | number | boolean {
  const trimmed = raw.trim();
  // Quoted string
  if (trimmed.startsWith("'") && trimmed.endsWith("'")) {
    return trimmed.slice(1, -1);
  }
  // Boolean
  if (trimmed.toLowerCase() === "true") return true;
  if (trimmed.toLowerCase() === "false") return false;
  // Number
  const num = Number(trimmed);
  if (!isNaN(num)) return num;
  return trimmed;
}

/** Compares a field value against a parsed OData value using the given operator */
function compareValues(
  fieldVal: unknown,
  op: string,
  compareVal: string | number | boolean,
): boolean {
  // Boolean comparison
  if (typeof compareVal === "boolean") {
    return op === "eq" ? Boolean(fieldVal) === compareVal : false;
  }
  // Numeric comparison
  if (typeof compareVal === "number") {
    const numField = Number(fieldVal);
    if (isNaN(numField)) return false;
    switch (op) {
      case "eq": return numField === compareVal;
      case "ge": return numField >= compareVal;
      case "le": return numField <= compareVal;
      case "gt": return numField > compareVal;
      case "lt": return numField < compareVal;
    }
  }
  // String comparison
  const strField = String(fieldVal ?? "");
  switch (op) {
    case "eq": return strField === compareVal;
    case "ge": return strField >= compareVal;
    case "le": return strField <= compareVal;
    case "gt": return strField > compareVal;
    case "lt": return strField < compareVal;
  }
  return true;
}
```

```ts
import { queryOptions, keepPreviousData } from "@tanstack/react-query";
import type { ApiResponse, BoxDto, BoxFilters } from "@/types/api";

const BASE = "/api/v1/box";

const STRING_FILTER_FIELDS = ["name", "objectCode"] as const;

function buildListParams(filters: BoxFilters): URLSearchParams {
  const { page = 1, pageSize = 20, ...rest } = filters;
  const params = new URLSearchParams();
  params.set("$skip", String((page - 1) * pageSize));
  params.set("$top", String(pageSize));

  const clauses: string[] = [];

  // String contains filters
  for (const f of STRING_FILTER_FIELDS) {
    if (rest[f]) {
      clauses.push(
        `contains_ignoring_case(${f},'${String(rest[f]).replace(/'/g, "''")}')`
      );
    }
  }

  // Tags — array search with any()
  if (rest.tags) {
    clauses.push(
      `tags/any(t: contains_ignoring_case(t,'${rest.tags.replace(/'/g, "''")}'))`
    );
  }

  // Shape — enum eq
  if (rest.shape) {
    clauses.push(`shape eq '${rest.shape}'`);
  }

  // Number range
  if (rest.numMin !== undefined) {
    clauses.push(`num ge ${rest.numMin}`);
  }
  if (rest.numMax !== undefined) {
    clauses.push(`num le ${rest.numMax}`);
  }

  // Boolean
  if (rest.checkbox !== undefined) {
    clauses.push(`checkbox eq ${rest.checkbox}`);
  }

  // Date range
  if (rest.dateStrFrom) {
    clauses.push(`dateStr ge '${rest.dateStrFrom}'`);
  }
  if (rest.dateStrTo) {
    clauses.push(`dateStr le '${rest.dateStrTo}'`);
  }

  // Relation: item (single)
  if (rest.itemId !== undefined) {
    clauses.push(`item/id eq ${rest.itemId}`);
  }

  // Relation: things (multi)
  if (rest.thingIds && rest.thingIds.length > 0) {
    const inner = rest.thingIds.map((id) => `t/id eq ${id}`).join(" or ");
    clauses.push(`things/any(t: ${inner})`);
  }

  if (clauses.length > 0) {
    params.set("$filter", clauses.join(" and "));
  }
  return params;
}

export async function fetchBoxes(
  filters: BoxFilters = {},
): Promise<ApiResponse<BoxDto[]>> {
  const params = buildListParams(filters);
  const response = await fetch(`${BASE}?${params}`);
  if (!response.ok) {
    throw new Error(`Failed to fetch boxes: ${response.status}`);
  }
  return response.json();
}

export async function fetchBox(
  id: number,
): Promise<ApiResponse<BoxDto>> {
  const response = await fetch(`${BASE}/${id}`);
  if (!response.ok) {
    throw new Error(`Failed to fetch box ${id}: ${response.status}`);
  }
  return response.json();
}

export async function patchBox(
  id: number,
  data: Partial<BoxDto>,
): Promise<ApiResponse<BoxDto>> {
  const response = await fetch(`${BASE}/${id}`, {
    method: "PATCH",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(data),
  });
  if (!response.ok) {
    throw new Error(`Failed to patch box ${id}: ${response.status}`);
  }
  return response.json();
}

export function boxesQueryOptions(filters: BoxFilters = {}) {
  return queryOptions({
    queryKey: ["boxes", "list", filters],
    queryFn: () => fetchBoxes(filters),
    placeholderData: keepPreviousData,
  });
}

  return queryOptions({
    queryKey: ["boxes", "detail", id],
    queryFn: () => fetchBox(id),
  });
}
```

```tsx
    shape: search.shape === "O" || search.shape === "X" ? search.shape : undefined,
    numMin: typeof search.numMin === "number" ? search.numMin : undefined,
    numMax: typeof search.numMax === "number" ? search.numMax : undefined,
    checkbox: typeof search.checkbox === "boolean" ? search.checkbox : undefined,
    dateStrFrom: typeof search.dateStrFrom === "string" ? search.dateStrFrom : undefined,
    dateStrTo: typeof search.dateStrTo === "string" ? search.dateStrTo : undefined,
    itemId: typeof search.itemId === "number" ? search.itemId : undefined,
    thingIds: typeof search.thingIds === "string"
      ? search.thingIds.split(",").map(Number).filter((n) => !isNaN(n))
      : Array.isArray(search.thingIds)
        ? (search.thingIds as number[])
        : undefined,
  };
}

export const Route = createFileRoute("/box/")({
  validateSearch,
  loaderDeps: ({ search }) => search,
  loader: ({ context, deps }) =>
    context.queryClient.ensureQueryData(boxesQueryOptions(deps)),
  component: BoxPage,
});

const columnHelper = createColumnHelper<BoxDto>();

const columns = [
  columnHelper.accessor("id",         { header: "ID",          size: 60  }),
  columnHelper.accessor("name",       { header: "Name",        size: 200 }),
  columnHelper.accessor("objectCode", { header: "Object Code", size: 130 }),
  columnHelper.accessor("shape",      { header: "Shape",       size: 80  }),
  columnHelper.accessor("num",        { header: "Num",         size: 100 }),
  columnHelper.accessor("item", {
    header: "Item",
    size: 180,
    cell: ({ getValue }) => getValue()?.name ?? "—",
  }),
  columnHelper.accessor("things", {
    header: "Things",
    size: 200,
    cell: ({ getValue }) => {
      const things = getValue();
      if (!things || things.length === 0) return "—";
      return things.map((t) => t.name).join(", ");
    },
  }),
  columnHelper.accessor("dateStr",  { header: "Date",     size: 120 }),
  columnHelper.accessor("checkbox", {
    header: "Checkbox",
    size: 90,
    cell: ({ getValue }) => (getValue() ? "Yes" : "No"),
  }),
  columnHelper.accessor("tags", {
    header: "Tags",
    size: 200,
    cell: ({ getValue }) => getValue()?.join(", ") ?? "",
  }),
];

const columnResizeMode: ColumnResizeMode = "onChange";

const boxSchema = z.object({
  name: z.string().min(1, "Name is required"),
  objectCode: z.string().nullable(),
  shape: z.enum(["O", "X"]),
  num: z.number(),
  dateStr: z.string(),
  checkbox: z.boolean(),
});

function BoxPage() {
  const { filters, setFilters, setPage } = useFilters("/box/");
  const { data } = useSuspenseQuery(boxesQueryOptions(filters));

  const page = filters.page ?? 1;
  const pageSize = filters.pageSize ?? 20;
  const pageCount = Math.ceil(data.count / pageSize);

  const table = useReactTable({
    data: data.data,
    columns,
    columnResizeMode,
    getCoreRowModel: getCoreRowModel(),
    manualFiltering: true,
    manualPagination: true,
    manualSorting: true,
    rowCount: data.count,
    state: {
      pagination: { pageIndex: page - 1, pageSize },
    },
  });

  const queryClient = useQueryClient();

  const updateMutation = useMutation({
    mutationFn: ({ id, data }: { id: number; data: Partial<BoxDto> }) =>
      patchBox(id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["boxes"] });
    },
  });

  const editConfig = useMemo<EditConfig<BoxDto>>(
    () => ({
      fields: {
        name: { type: "text" },
        objectCode: { type: "text" },
        shape: {
          type: "select",
          options: [
            { label: "O", value: "O" },
            { label: "X", value: "X" },
          ],
        },
        num: { type: "number" },
        dateStr: { type: "text" },
        checkbox: { type: "checkbox" },
      },
      disabledFields: ["id"],
      schema: boxSchema,
      onSave: async (id, data) => {
        await updateMutation.mutateAsync({ id: id as number, data });
      },
      rowId: (row) => row.id,
    }),
    [updateMutation],
  );

  // Hydrate selected relation objects from query cache so the filter UI
  // shows the correct labels on page reload (when only IDs are in the URL).
  const { data: allItems } = useQuery({
    ...itemsQueryOptions(),
    enabled: filters.itemId !== undefined,
  });
  const { data: allThings } = useQuery({
    ...thingsQueryOptions(),
    enabled: !!filters.thingIds && filters.thingIds.length > 0,
  });

  const selectedItem = useMemo<ItemDto | undefined>(() => {
    if (filters.itemId === undefined || !allItems) return undefined;
    return allItems.data.find((i) => i.id === filters.itemId);
  }, [filters.itemId, allItems]);

  const selectedThings = useMemo<ThingDto[]>(() => {
    if (!filters.thingIds || filters.thingIds.length === 0 || !allThings) return [];
    const idSet = new Set(filters.thingIds);
    return allThings.data.filter((t) => idSet.has(t.id));
  }, [filters.thingIds, allThings]);

  return (
    <div>
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold">Boxes</h1>
          <p className="mt-2 text-muted-foreground">{data.count} boxes</p>
        </div>
      </div>

      <SimpleTable
        table={table}
        page={page}
        pageCount={pageCount}
        onPageChange={setPage}
        filterSlot={
          <FilterBar>
            <TextFilter
              placeholder="Name"
              value={filters.name}
              onChange={(v) => setFilters({ name: v })}
            />
            <TextFilter
              placeholder="Object Code"
              value={filters.objectCode}
              onChange={(v) => setFilters({ objectCode: v })}
            />
            <SelectFilter
              placeholder="Shape"
              options={[
                { label: "O", value: "O" },
                { label: "X", value: "X" },
              ]}
              value={filters.shape}
              onChange={(v) => setFilters({ shape: v as "O" | "X" | undefined })}
            />
            <CheckboxFilter
              label="Checkbox"
              value={filters.checkbox}
              onChange={(v) => setFilters({ checkbox: v })}
            />
            <NumberRangeFilter
              placeholderMin="Num min"
              placeholderMax="Num max"
              valueMin={filters.numMin}
              valueMax={filters.numMax}
              onChangeMin={(v) => setFilters({ numMin: v })}
              onChangeMax={(v) => setFilters({ numMax: v })}
            />
            <TextFilter
              placeholder="Tags"
              value={filters.tags}
              onChange={(v) => setFilters({ tags: v })}
            />
            <DateRangeFilter
              className="col-span-2"
              placeholderFrom="Date from"
              placeholderTo="Date to"
              valueFrom={filters.dateStrFrom}
              valueTo={filters.dateStrTo}
              onChangeFrom={(v) => setFilters({ dateStrFrom: v })}
              onChangeTo={(v) => setFilters({ dateStrTo: v })}
            />
            <RelationFilter<ItemDto>
              className="col-span-2"
              mode="both"
              multi={false}
              value={selectedItem}
              onChange={(v) => {
                const item = v as ItemDto | undefined;
                setFilters({ itemId: item ? item.id : undefined });
              }}
              queryFn={fetchItems}
              queryKey={["items", "all"]}
              getLabel={(i) => i.name}
              getId={(i) => i.id}
            />
            <RelationFilter<ThingDto>
              className="col-span-2"
              mode="both"
              multi={true}
              value={selectedThings.length > 0 ? selectedThings : undefined}
              onChange={(v) => {
                const things = v as ThingDto[] | undefined;
                setFilters({
                  thingIds: things && things.length > 0
                    ? things.map((t) => t.id)
                    : undefined,
                });
              }}
              queryFn={fetchThings}
              queryKey={["things", "all"]}
              getLabel={(t) => t.name}
              getId={(t) => t.id}
            />
          </FilterBar>
        }
        editConfig={editConfig}
        pinnedLeftColumnId="name"
      />
    </div>
  );
}
```

Note: The `selectedItem` / `selectedThings` memos hydrate from the items/things queries so that on page reload (when only IDs are in the URL), the RelationFilter shows the correct labels.
